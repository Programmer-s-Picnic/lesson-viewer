Step 2 — BUILD (Analyzer logic)

Normalize text, match keywords, compute results. 
def normalize(text: str) -> str:
    return " ".join(text.lower().split())

def analyze(resume_text: str, skills: list[str]):
    t = normalize(resume_text)
    matched = []
    missing = []
    counts = {}

    for sk in skills:
        c = t.count(sk.lower())
        counts[sk] = c
        if c > 0:
            matched.append(sk)
        else:
            missing.append(sk)

    total = len(skills)
    score = (len(matched) / total) * 100 if total else 0

    suggestions = missing[:5]  # top 5 missing
    return score, matched, missing, counts, suggestions

def pretty_print(score, matched, missing, counts, suggestions):
    print("\n=== Resume Analyzer (Mini) ===")
    print(f"Score: {score:.1f}%")
    print("\nMatched skills:", ", ".join(matched) if matched else "None")
    print("Missing skills:", ", ".join(missing) if missing else "None")

    print("\nKeyword counts:")
    for k, v in counts.items():
        print(f"- {k}: {v}")

    print("\nSuggestions to improve:")
    for s in suggestions:
        print(f"* Add projects / lines showing: {s}")

def main():
    skills = ["python","loops","functions","lists","dictionaries","oops","sql","pandas","html","css","javascript"]

    print("Paste resume text (end with a blank line):")
    lines = []
    while True:
        line = input()
        if line.strip() == "":
            break
        lines.append(line)

    resume_text = "\n".join(lines).strip()
    if not resume_text:
        print("No text provided.")
        return

    score, matched, missing, counts, suggestions = analyze(resume_text, skills)
    pretty_print(score, matched, missing, counts, suggestions)

if __name__ == "__main__":
    main()
